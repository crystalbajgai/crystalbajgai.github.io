# Python Program to make a simple calculator # take inputs num1 = int(input("Enter first number:")) num2 = int(input("Enter second number: ")) # operation choice print("Operations: +, -, /, *") select = input("Select operation: ") # check operation and display result # adds numbers if select == "+": print(num1, "+", num2, "=", num1+num2) # subtracts numbers elif select == "-": print(num1, "-", num2, "=", num1-num2) # divides numbers elif select == "/": print(num1, "/", num2, "=", num1/num2) # multiplies numbers elif select == "*": print(num1, "*", num2, "=", num1*num2) # if the operator is invalid else: print('Invalid Input')
